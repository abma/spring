# Spring supplied CMake build file

SET(TUIO_SRC_ROOT "${ENGINE_SRC_ROOT_DIR}/lib/tuio")

SET(tuioSources
		"TuioClient"
		"TuioTime"
	)

MESSAGE(STATUS "OSCPACK_SRC_ROOT: " ${OSCPACK_SRC_ROOT})

INCLUDE_DIRECTORIES(${OSCPACK_SRC_ROOT} ${OSCPACK_SRC_ROOT}/ip)
ADD_LIBRARY(tuiocpp STATIC EXCLUDE_FROM_ALL ${tuioSources})
ADD_DEPENDENCIES(tuiocpp oscpack)
TARGET_LINK_LIBRARIES(tuiocpp oscpack)

IF (APPLE)
	SET_TARGET_PROPERTIES(tuiocpp PROPERTIES COMPILE_FLAGS -DOSC_HOST_BIG_ENDIAN)
ELSE (APPLE)
	SET_TARGET_PROPERTIES(tuiocpp PROPERTIES COMPILE_FLAGS -DOSC_HOST_LITTLE_ENDIAN)
ENDIF (APPLE)
